⚠️ 無法連線至後台

`; } } // 新增功能：根據試算表設定，自動畫出右側座位圖 function renderFloorPlan() { const floorPlanContainer = document.getElementById("floor-plan"); floorPlanContainer.innerHTML = ""; // 先清空原本寫死的座位 // 顏色陣列，讓不同桌次有不同的漂亮標籤顏色 const colors = ['text-rose-500 bg-rose-50', 'text-blue-500 bg-blue-50', 'text-purple-500 bg-purple-50', 'text-amber-500 bg-amber-50', 'text-emerald-500 bg-emerald-50', 'text-teal-500 bg-teal-50', 'text-indigo-500 bg-indigo-50']; tables.forEach((t, index) => { const colorClass = colors[index % colors.length]; const tableDiv = document.createElement("div"); tableDiv.id = `table-${t.table_id}`; tableDiv.className = "border-2 border-stone-300 rounded-full aspect-square flex flex-col items-center justify-center bg-stone-50 transition-all duration-300 p-2 text-center"; tableDiv.innerHTML = ` ${t.table_id} 桌 ${t.table_name} `; floorPlanContainer.appendChild(tableDiv); }); } function renderDashboard() { const total = guests.length; const checked = guests.filter(g => g.status === "已報到").length; const unchecked = total - checked; const percentage = total > 0 ? Math.round((checked / total) * 100) : 0; document.getElementById("total-count").innerText = total; document.getElementById("checked-count").innerText = checked; document.getElementById("unchecked-count").innerText = unchecked; document.getElementById("progress-bar").style.width = `${percentage}%`; document.getElementById("progress-text").innerText = `目前報到率：${percentage}%`; } function renderList(data) { const listContainer = document.getElementById("guest-list"); listContainer.innerHTML = ""; if (data.length === 0) { listContainer.innerHTML = `

`; return; } data.forEach(guest => { const item = document.createElement("div"); item.className = "flex items-center justify-between py-4 transition-colors duration-200"; const isChecked = guest.status === "已報到"; // 從 tables 對照表找出這一桌的中文字名稱 const tObj = tables.find(t => t.table_id == guest.table); const tableNameText = tObj ? tObj.table_name : `第 ${guest.table} 桌`; item.innerHTML = `

${isChecked ? `✓ 已報到` : `` }
`; listContainer.appendChild(item); }); } async function checkIn(id) { const guest = guests.find(g => g.id === id); if (!guest) return; guest.status = "已報到"; renderDashboard(); renderList(guests); try { const response = await fetch(API_URL, { method: "POST", body: JSON.stringify({ id: id }) }); const result = await response.json(); if (result.success) { document.getElementById("search-input").value = ""; const tObj = tables.find(t => t.table_id == guest.table); const tableNameText = tObj ? tObj.table_name : `${guest.table}桌`; document.getElementById("modal-name").innerText = guest.name; document.getElementById("modal-table").innerText = `${guest.table}桌 (${tableNameText})`; const modal = document.getElementById("success-modal"); modal.classList.remove("hidden"); modal.classList.add("flex"); highlightTable(guest.table); } else { alert("後台更新失敗：" + result.message); await fetchGuests(); } } catch (error) { console.error("報到請求失敗:", error); alert("網路連線異常，簽到資料可能未同步，請重試！"); await fetchGuests(); } } function highlightTable(tableName) { document.querySelectorAll("#floor-plan > div").forEach(el => el.classList.remove("table-active")); const targetTable = document.getElementById(`table-${tableName}`); if (targetTable) { targetTable.classList.add("table-active"); } } function closeModal() { const modal = document.getElementById("success-modal"); modal.classList.add("hidden"); modal.classList.remove("flex"); }
